1//! Planner command handler - CSV metrics export
2
3use nu_analytics::{error, info};
4use nu_analytics::core::{
5    metrics, metrics_export,
6    models::{Degree, Plan},
7    planner::parse_curriculum_csv,
8};
9use std::path::{Path, PathBuf};
10
11/// Run CSV export for a single input file
12///
13/// # Arguments
14/// * `input_file` - Path to input CSV file
15/// * `output_file` - Optional explicit output path
16/// * `metrics_dir` - Directory for output when `output_file` is None
17/// * `verbose` - Whether to show detailed metrics output
18pub fn run_single(input_file: &Path, output_file: Option<&Path>, metrics_dir: &str, verbose: bool) {
19    if let Err(err) = export_csv(input_file, output_file, metrics_dir, verbose) {
20        error!("Planner failed for {}: {err}", input_file.display());
21        eprintln!("{err}");
22    }
23}
24
25fn export_csv(
26    input_file: &Path,
27    output_file: Option<&Path>,
28    metrics_dir: &str,
29    verbose: bool,
30) -> Result<(), String> {
31    let school = parse_curriculum_csv(input_file).map_err(|e| {
32        error!("Failed to load curriculum {}: {e}", input_file.display());
33        format!("✗ Failed to load {}: {e}", input_file.display())
34    })?;
35
36    if verbose {
37        println!(
38            "✓ Curriculum loaded successfully from: {}",
39            input_file.display()
40        );
41    } else {
42        info!("Curriculum loaded: {}", input_file.display());
43    }
44
45    let dag = school.build_dag();
46
47    let all_metrics = metrics::compute_all_metrics(&dag).map_err(|e| {
48        error!(
49            "Metrics computation failed for {}: {e}",
50            input_file.display()
51        );
52        format!(
53            "✗ Failed to compute metrics for {}: {e}",
54            input_file.display()
55        )
56    })?;
57
58    let plan = if let Some(p) = school.plans.first() {
59        p.clone()
60    } else {
61        // If no explicit plans are defined, create a default plan that includes all courses.
62        // This ensures metrics can be computed for the entire curriculum even if individual
63        // plans haven't been specified in the input file. The default plan is named "All Courses"
64        // and associated with the first degree in the school (if any).
65        let mut default_plan = Plan::new(
66            "All Courses".to_string(),
67            school.degrees.first().map_or_else(String::new, Degree::id),
68        );
69        for course in &dag.courses {
70            default_plan.add_course(course.clone());
71        }
72        default_plan
73    };
74
75    let final_output_path: PathBuf = if let Some(output) = output_file {
76        // Ensure parent directory exists
77        if let Some(parent) = output.parent() {
78            std::fs::create_dir_all(parent).map_err(|e| {
79                format!(
80                    "✗ Failed to create output directory {}: {e}",
81                    parent.display()
82                )
83            })?;
84        }
85        output.to_path_buf()
86    } else {
87        let metrics_path = PathBuf::from(metrics_dir);
88        std::fs::create_dir_all(&metrics_path).map_err(|e| {
89            format!(
90                "✗ Failed to create metrics directory {}: {e}",
91                metrics_path.display()
92            )
93        })?;
94
95        let filename = input_file
96            .file_stem()
97            .and_then(|stem| stem.to_str())
98            .unwrap_or("curriculum")
99            .to_string();
100        let output_filename = format!("{filename}_w_metrics.csv");
101        metrics_path.join(output_filename)
102    };
103
104    let plan_name = plan.name.clone();
105    let degree_label = school
106        .degrees
107        .first()
108        .map_or_else(|| "Unknown Degree".to_string(), Degree::id);
109    let institution_label = plan
110        .institution
111        .clone()
112        .unwrap_or_else(|| school.name.clone());
113
114    match metrics_export::export_metrics_csv(&school, &plan, &all_metrics, &final_output_path) {
115        Ok(summary) => {
116            println!("✓ Metrics exported to: {}", final_output_path.display());
117            info!(
118                "Exported curriculum metrics to: {}",
119                final_output_path.display()
120            );
121
122            if verbose {
123                println!("\n=== Plan Summary for {plan_name} ({degree_label}) at {institution_label} ===");
124                println!("Total Structural Complexity: {}", summary.total_complexity);
125                println!(
126                    "Longest Delay: {} ({})",
127                    summary.longest_delay,
128                    summary.longest_delay_path.join("->")
129                );
130                println!(
131                    "Highest Centrality: {} ({})",
132                    summary.highest_centrality, summary.highest_centrality_course
133                );
134            }
135            Ok(())
136        }
137        Err(e) => Err(format!(
138            "✗ Failed to export metrics to {}: {e}",
139            final_output_path.display()
140        )),
141    }
142}
