12/// Run the planner command for one or more input files.
13///
14/// # Arguments
15/// * `input_files` - Paths to input CSV files
16/// * `output_files` - Optional output paths; must match inputs 1:1 when provided
17/// * `config` - Configuration containing default output directory
18/// * `verbose` - Whether to show detailed metrics output
19pub fn run(input_files: &[PathBuf], output_files: &[PathBuf], config: &Config, verbose: bool) {
20    if input_files.is_empty() {
21        eprintln!("✗ No input files provided.");
22        return;
23    }
24
25    if !output_files.is_empty() && output_files.len() != input_files.len() {
26        eprintln!(
27            "✗ When using -o/--output, provide one output path per input file ({} inputs, {} outputs).",
28            input_files.len(),
29            output_files.len()
30        );
31        return;
32    }
33
34    for (idx, input_file) in input_files.iter().enumerate() {
35        let output_file = output_files.get(idx).map(PathBuf::as_path);
36        if let Err(err) = export_single(input_file, output_file, config, verbose) {
37            error!("Planner failed for {}: {err}", input_file.display());
38            eprintln!("{err}");
39        }
40    }
41}
42
43fn export_single(
44    input_file: &Path,
45    output_file: Option<&Path>,
46    config: &Config,
47    verbose: bool,
48) -> Result<(), String> {
49    let school = parse_curriculum_csv(input_file).map_err(|e| {
50        error!("Failed to load curriculum {}: {e}", input_file.display());
51        format!("✗ Failed to load {}: {e}", input_file.display())
52    })?;
53
54    if verbose {
55        println!(
56            "✓ Curriculum loaded successfully from: {}",
57            input_file.display()
58        );
59    } else {
60        info!("Curriculum loaded: {}", input_file.display());
61    }
62
63    let dag = school.build_dag();
64
65    let all_metrics = metrics::compute_all_metrics(&dag).map_err(|e| {
66        error!(
67            "Metrics computation failed for {}: {e}",
68            input_file.display()
69        );
70        format!(
71            "✗ Failed to compute metrics for {}: {e}",
72            input_file.display()
73        )
74    })?;
75
76    let plan = if let Some(p) = school.plans.first() {
77        p.clone()
78    } else {
79        // If no explicit plans are defined, create a default plan that includes all courses.
80        // This ensures metrics can be computed for the entire curriculum even if individual
81        // plans haven't been specified in the input file. The default plan is named "All Courses"
82        // and associated with the first degree in the school (if any).
83        let mut default_plan = Plan::new(
84            "All Courses".to_string(),
85            school.degrees.first().map_or_else(String::new, Degree::id),
86        );
87        for course in &dag.courses {
88            default_plan.add_course(course.clone());
89        }
90        default_plan
91    };
92
93    let final_output_path: PathBuf = if let Some(output) = output_file {
94        output.to_path_buf()
95    } else {
96        let out_dir = PathBuf::from(&config.paths.out_dir);
97        std::fs::create_dir_all(&out_dir).map_err(|e| {
98            format!(
99                "✗ Failed to create output directory {}: {e}",
100                out_dir.display()
101            )
102        })?;
103
104        let filename = input_file
105            .file_stem()
106            .and_then(|stem| stem.to_str())
107            .unwrap_or("curriculum")
108            .to_string();
109        let output_filename = format!("{filename}_w_metrics.csv");
110        out_dir.join(output_filename)
111    };
112
113    let plan_name = plan.name.clone();
114    let degree_label = school
115        .degrees
116        .first()
117        .map_or_else(|| "Unknown Degree".to_string(), Degree::id);
118    let institution_label = plan
119        .institution
120        .clone()
121        .unwrap_or_else(|| school.name.clone());
122
123    match metrics_export::export_metrics_csv(&school, &plan, &all_metrics, &final_output_path) {
124        Ok(summary) => {
125            println!("✓ Metrics exported to: {}", final_output_path.display());
126            info!(
127                "Exported curriculum metrics to: {}",
128                final_output_path.display()
129            );
130
131            if verbose {
132                println!("\n=== Plan Summary for {plan_name} ({degree_label}) at {institution_label} ===");
133                println!("Total Structural Complexity: {}", summary.total_complexity);
134                println!(
135                    "Longest Delay: {} ({})",
136                    summary.longest_delay,
137                    summary.longest_delay_path.join("->")
138                );
139                println!(
140                    "Highest Centrality: {} ({})",
141                    summary.highest_centrality, summary.highest_centrality_course
142                );
143            }
144            Ok(())
145        }
146        Err(e) => Err(format!(
147            "✗ Failed to export metrics to {}: {e}",
148            final_output_path.display()
149        )),
150    }
151}
