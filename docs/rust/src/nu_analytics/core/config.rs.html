99    /// Merge missing fields from defaults into this config
100    ///
101    /// This method is used when loading configuration to ensure that newly added
102    /// configuration fields are populated with their default values. Only fields
103    /// that are empty in the current config and non-empty in defaults are updated.
104    ///
105    /// # Returns
106    ///
107    /// `true` if any fields were added/changed, `false` otherwise
108    ///
109    /// # Examples
110    ///
111    /// ```ignore
112    /// let mut config = Config::from_toml(old_config_str)?;
113    /// let defaults = Config::from_defaults();
114    /// if config.merge_defaults(&defaults) {
115    ///     // Config was updated with new fields
116    ///     config.save()?;
117    /// }
118    /// ```
119    #[allow(clippy::useless_let_if_seq)]
120    pub fn merge_defaults(&mut self, defaults: &Self) -> bool {
121        let mut changed = false;
122
123        // Merge logging fields - only if they're empty (use defaults for empty values)
124        if self.logging.level.is_empty() && !defaults.logging.level.is_empty() {
125            self.logging.level.clone_from(&defaults.logging.level);
126            changed = true;
127        }
128        if self.logging.file.is_empty() && !defaults.logging.file.is_empty() {
129            self.logging.file.clone_from(&defaults.logging.file);
130            changed = true;
131        }
132
133        // Merge database fields - only add if default is non-empty
134        if self.database.token.is_empty() && !defaults.database.token.is_empty() {
135            self.database.token.clone_from(&defaults.database.token);
136            changed = true;
137        }
138        if self.database.endpoint.is_empty() && !defaults.database.endpoint.is_empty() {
139            self.database
140                .endpoint
141                .clone_from(&defaults.database.endpoint);
142            changed = true;
143        }
144
145        // Merge paths fields
146        if self.paths.out_dir.is_empty() && !defaults.paths.out_dir.is_empty() {
147            self.paths.out_dir.clone_from(&defaults.paths.out_dir);
148            changed = true;
149        }
150
151        changed
152    }

154    /// Apply CLI-provided overrides onto the loaded configuration
155    ///
156    /// This allows command-line arguments to override configuration file values
157    /// without modifying the persistent configuration file. Only non-`None` values
158    /// in the overrides struct will replace config values.
159    ///
160    /// # Arguments
161    ///
162    /// * `overrides` - A `ConfigOverrides` struct with optional override values
163    ///
164    /// # Examples
165    ///
166    /// ```ignore
167    /// let mut config = Config::load();
168    /// let overrides = ConfigOverrides {
169    ///     level: Some("debug".to_string()),
170    ///     ..Default::default()
171    /// };
172    /// config.apply_overrides(&overrides);
173    /// // config.logging.level is now "debug" for this run only
174    /// ```
175    pub fn apply_overrides(&mut self, overrides: &ConfigOverrides) {
176        if let Some(level) = &overrides.level {
177            self.logging.level.clone_from(level);
178        }
179        if let Some(file) = &overrides.file {
180            self.logging.file.clone_from(file);
181        }
182        if let Some(verbose) = overrides.verbose {
183            self.logging.verbose = verbose;
184        }
185
186        if let Some(token) = &overrides.db_token {
187            self.database.token.clone_from(token);
188        }
189        if let Some(endpoint) = &overrides.db_endpoint {
190            self.database.endpoint.clone_from(endpoint);
191        }
192
193        if let Some(out_dir) = &overrides.out_dir {
194            self.paths.out_dir.clone_from(out_dir);
195        }
196    }

212    /// Expand `$NU_ANALYTICS` variable in a string
213    ///
214    /// Replaces occurrences of `$NU_ANALYTICS` with the actual nuanalytics
215    /// directory path. This allows configuration values to reference the
216    /// config directory dynamically.
217    ///
218    /// # Arguments
219    ///
220    /// * `value` - The string potentially containing `$NU_ANALYTICS`
221    ///
222    /// # Returns
223    ///
224    /// The string with `$NU_ANALYTICS` expanded to the actual path
225    ///
226    /// # Examples
227    ///
228    /// ```ignore
229    /// let expanded = Config::expand_variables("$NU_ANALYTICS/logs/app.log");
230    /// // Returns something like "/home/user/.config/nuanalytics/logs/app.log"
231    /// ```
232    #[must_use]
233    fn expand_variables(value: &str) -> String {
234        if value.contains("$NU_ANALYTICS") {
235            let nu_analytics_dir = Self::get_nuanalytics_dir();
236            value.replace("$NU_ANALYTICS", nu_analytics_dir.to_str().unwrap_or("."))
237        } else {
238            value.to_string()
239        }
240    }

242    /// Initialize config from a TOML string
243    ///
244    /// Parses a TOML configuration string and expands any `$NU_ANALYTICS` variables
245    /// in the values. Missing fields will use their serde defaults (typically empty
246    /// strings or false).
247    ///
248    /// # Arguments
249    ///
250    /// * `toml_str` - A TOML-formatted configuration string
251    ///
252    /// # Errors
253    ///
254    /// Returns an error if the TOML cannot be parsed or doesn't match the expected schema
255    ///
256    /// # Examples
257    ///
258    /// ```ignore
259    /// let config = Config::from_toml(r#"
260    /// [Logging]
261    /// level = "info"
262    /// file = "$NU_ANALYTICS/app.log"
263    /// "#)?;
264    /// ```
265    pub fn from_toml(toml_str: &str) -> Result<Self, toml::de::Error> {
266        let mut config: Self = toml::from_str(toml_str)?;
267
268        // Expand variables in config values
269        config.logging.file = Self::expand_variables(&config.logging.file);
270        config.database.token = Self::expand_variables(&config.database.token);
271        config.database.endpoint = Self::expand_variables(&config.database.endpoint);
272        config.paths.out_dir = Self::expand_variables(&config.paths.out_dir);
273
274        Ok(config)
275    }

279    /// Loads the compiled-in default configuration that is bundled with the binary.

299    }

301    /// Load configuration from file, or create from defaults if not found
302    ///
303    /// This is the primary way to load configuration. It handles several scenarios:
304    /// - If config file exists: Loads from file, merges missing fields from defaults, saves updated config
305    /// - If config file doesn't exist (first run): Creates config directory if needed, loads defaults, saves to file
306    ///
307    /// The merge behavior ensures that upgrading the application automatically adds new config
308    /// fields while preserving existing user settings.
309    ///
310    /// # Returns
311    /// A `Config` instance loaded from file or defaults. Falls back to defaults if any error occurs
312    /// during loading.
313    ///
314    /// # Examples
315    /// ```ignore
316    /// let config = Config::load();
317    /// // Config is now loaded from ~/.config/nuanalytics/config.toml (or defaults if first run)
318    /// ```
319    #[must_use]
320    pub fn load() -> Self {
321        let config_file = Self::get_config_file_path();
322        let defaults = Self::from_defaults();
323
324        if config_file.exists() {
325            if let Ok(content) = fs::read_to_string(&config_file) {
326                if let Ok(mut config) = Self::from_toml(&content) {
327                    // Merge any missing fields from defaults
328                    if config.merge_defaults(&defaults) {
329                        // Save the updated config with new fields
330                        let _ = config.save();
331                    }
332                    return config;
333                }
334            }
335        } else {
336            // First run: create directory and config file from defaults
337
338            // Create the directory if it doesn't exist
339            if let Some(parent) = config_file.parent() {
340                let _ = fs::create_dir_all(parent);
341            }
342
343            // Save the default config
344            let _ = defaults.save();
345
346            return defaults;
347        }
348
349        defaults
350    }

352    /// Save configuration to file
353    ///
354    /// Serializes the current configuration to TOML format and writes it to the
355    /// platform-specific config file. The config directory will be created if it
356    /// doesn't exist.
357    ///
358    /// The saved file will use the format:
359    /// ```toml
360    /// [Logging]
361    /// level = "info"
362    /// file = "$NU_ANALYTICS/logs/nuanalytics.log"
363    /// verbose = false
364    ///
365    /// [Database]
366    /// token = "your-token"
367    /// endpoint = "https://api.example.com"
368    ///
369    /// [Paths]
370    /// out_dir = "$NU_ANALYTICS/output"
371    /// ```
372    ///
373    /// # Errors
374    /// Returns an error if:
375    /// - The config cannot be serialized to TOML (shouldn't happen)
376    /// - The config directory cannot be created
377    /// - The file cannot be written (permissions, disk full, etc.)
378    ///
379    /// # Examples
380    /// ```ignore
381    /// let mut config = Config::load()?;
382    /// config.logging.level = "debug".to_string();
383    /// config.save()?;
384    /// ```
385    pub fn save(&self) -> Result<(), Box<dyn std::error::Error>> {
386        let config_file = Self::get_config_file_path();
387        if let Some(parent) = config_file.parent() {
388            fs::create_dir_all(parent)?;
389        }
390        let toml_str = toml::to_string_pretty(self)?;
391        fs::write(&config_file, toml_str)?;
392        Ok(())
393    }

395    /// Get a configuration value by key
396    ///
397    /// Retrieves a configuration value using a string key that maps to the config structure.
398    /// Supports all config fields in the format `section.field` or just `field` for top-level fields.
399    ///
400    /// Supported keys:
401    /// - `level`: Logging level ("debug", "info", "warn", "error")
402    /// - `file`: Log file path
403    /// - `verbose`: Verbose logging boolean
404    /// - `token`: Database authentication token
405    /// - `endpoint`: Database API endpoint
406    /// - `out_dir`: Output directory path
407    ///
408    /// # Arguments
409    /// - `key`: The configuration key to retrieve
410    ///
411    /// # Returns
412    /// - `Some(String)`: The configuration value as a string
413    /// - `None`: If the key is not recognized
414    ///
415    /// # Examples
416    /// ```ignore
417    /// let config = Config::load()?;
418    /// if let Some(level) = config.get("level") {
419    ///     println!("Current log level: {}", level);
420    /// }
421    /// ```
422    #[must_use]
423    pub fn get(&self, key: &str) -> Option<String> {
424        match key {
425            "level" => Some(self.logging.level.clone()),
426            "file" => Some(self.logging.file.clone()),
427            "verbose" => Some(self.logging.verbose.to_string()),
428            "token" => Some(self.database.token.clone()),
429            "endpoint" => Some(self.database.endpoint.clone()),
430            "out_dir" => Some(self.paths.out_dir.clone()),
431            _ => None,
432        }
433    }

435    /// Set a configuration value by key
436    ///
437    /// Updates a configuration value using a string key and value. The value will be
438    /// validated and converted to the appropriate type.
439    ///
440    /// Supported keys and their value formats:
441    /// - `level`: String ("debug", "info", "warn", "error", "trace", "off")
442    /// - `file`: String (file path, can include `$NU_ANALYTICS`)
443    /// - `verbose`: Boolean ("true" or "false")
444    /// - `token`: String (any value)
445    /// - `endpoint`: String (typically a URL)
446    /// - `out_dir`: String (directory path)
447    ///
448    /// Note: This method updates the in-memory config. Call [`save()`](Config::save) to persist changes.
449    ///
450    /// # Arguments
451    /// - `key`: The configuration key to set
452    /// - `value`: The new value as a string
453    ///
454    /// # Errors
455    /// Returns an error if:
456    /// - The key is not recognized
457    /// - The value cannot be parsed (e.g., "maybe" for verbose boolean)
458    ///
459    /// # Examples
460    /// ```ignore
461    /// let mut config = Config::load()?;
462    /// config.set("level", "debug")?;
463    /// config.set("verbose", "true")?;
464    /// config.save()?;
465    /// ```
466    pub fn set(&mut self, key: &str, value: &str) -> Result<(), String> {
467        match key {
468            "level" => self.logging.level = value.to_string(),
469            "file" => self.logging.file = value.to_string(),
470            "verbose" => {
471                self.logging.verbose = value
472                    .parse::<bool>()
473                    .map_err(|_| format!("Invalid boolean value for 'verbose': '{value}'"))?;
474            }
475            "token" => self.database.token = value.to_string(),
476            "endpoint" => self.database.endpoint = value.to_string(),
477            "out_dir" => self.paths.out_dir = value.to_string(),
478            _ => return Err(format!("Unknown config key: '{key}'")),
479        }
480        Ok(())
481    }

483    /// Unset a configuration value by key (reset to default)
484    ///
485    /// Resets a single configuration value to its default value. This is useful for
486    /// reverting individual settings without losing all customizations.
487    ///
488    /// The default value is taken from the provided defaults config (typically from
489    /// [`from_defaults()`](Config::from_defaults)).
490    ///
491    /// Note: This method updates the in-memory config. Call [`save()`](Config::save) to persist changes.
492    ///
493    /// # Arguments
494    /// - `key`: The configuration key to reset
495    /// - `defaults`: A config instance containing default values
496    ///
497    /// # Errors
498    /// Returns an error if the key is not recognized.
499    ///
500    /// # Examples
501    /// ```ignore
502    /// let mut config = Config::load()?;
503    /// let defaults = Config::from_defaults();
504    ///
505    /// config.set("level", "trace")?;
506    /// config.unset("level", &defaults)?;  // Resets to "info"
507    /// config.save()?;
508    /// ```
509    pub fn unset(&mut self, key: &str, defaults: &Self) -> Result<(), String> {
510        match key {
511            "level" => self.logging.level.clone_from(&defaults.logging.level),
512            "file" => self.logging.file.clone_from(&defaults.logging.file),
513            "verbose" => self.logging.verbose = defaults.logging.verbose,
514            "token" => self.database.token.clone_from(&defaults.database.token),
515            "endpoint" => self
516                .database
517                .endpoint
518                .clone_from(&defaults.database.endpoint),
519            "out_dir" => self.paths.out_dir.clone_from(&defaults.paths.out_dir),
520            _ => return Err(format!("Unknown config key: '{key}'")),
521        }
522        Ok(())
523    }

525    /// Reset all configuration to defaults
526    ///
527    /// Deletes the configuration file, causing the next [`load()`](Config::load) call to
528    /// recreate it from defaults. This is a destructive operation that removes all user
529    /// customizations.
530    ///
531    /// If the config file doesn't exist, this method succeeds without doing anything.
532    ///
533    /// # Safety
534    /// This is a destructive operation. The CLI typically requires user confirmation
535    /// before calling this method.
536    ///
537    /// # Errors
538    /// Returns an error if:
539    /// - The config file exists but cannot be deleted (permissions, file locked, etc.)
540    ///
541    /// # Examples
542    /// ```ignore
543    /// // Typically preceded by user confirmation
544    /// Config::reset()?;
545    /// println!("Configuration reset to defaults");
546    ///
547    /// // Next load will recreate from defaults
548    /// let config = Config::load()?;
549    /// ```
550    pub fn reset() -> Result<(), std::io::Error> {
551        let config_file = Self::get_config_file_path();
552        if config_file.exists() {
553            fs::remove_file(config_file)?;
554        }
555        Ok(())
556    }

561        writeln!(f, "[logging]")?;
562        writeln!(f, "  level = \"{}\"", self.logging.level)?;
563        writeln!(f, "  file = \"{}\"", self.logging.file)?;
564        writeln!(f, "  verbose = {}", self.logging.verbose)?;

566        writeln!(f, "\n[database]")?;
567        writeln!(f, "  token = \"{}\"", self.database.token)?;
568        writeln!(f, "  endpoint = \"{}\"", self.database.endpoint)?;
569
570        writeln!(f, "\n[paths]")?;
571        writeln!(f, "  out_dir = \"{}\"", self.paths.out_dir)?;
572
573        Ok(())
574    }
