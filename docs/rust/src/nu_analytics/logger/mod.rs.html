24/// Logging levels.
25#[derive(Copy, Clone, Debug, PartialEq, Eq)]
26pub enum Level {
27    /// Error-level messages (always enabled).
28    Error = 1,
29    /// Warning-level messages (always enabled).
30    Warn = 2,
31    /// Info-level messages (requires `log-info` feature).
32    Info = 3,
33    /// Debug-level messages (requires `log-debug` feature and runtime enablement).
34    Debug = 4,
35}
36
37const fn default_level() -> u8 {
38    if cfg!(feature = "log-debug") {
39        Level::Debug as u8
40    } else if cfg!(feature = "log-info") {
41        Level::Info as u8
42    } else {
43        Level::Warn as u8
44    }
45}
46
47/// Global storage for the current log level.
48static LOG_LEVEL: LazyLock<AtomicU8> = LazyLock::new(|| AtomicU8::new(default_level()));
49#[cfg(feature = "log-debug")]
50static DEBUG_ENABLED: AtomicBool = AtomicBool::new(true);
51#[cfg(feature = "verbose")]
52static VERBOSE_ENABLED: AtomicBool = AtomicBool::new(false);
53#[cfg(feature = "file-logging")]
54static LOG_FILE: LazyLock<Mutex<Option<File>>> = LazyLock::new(|| Mutex::new(None));
55
56/// Set the global log level.
57pub fn set_level(level: Level) {
58    LOG_LEVEL.store(level as u8, Ordering::SeqCst);
59}
60
61#[must_use]
62/// Parse level from string (case-insensitive) and set it. Returns `true` on success.
63pub fn set_level_from_str(level: &str) -> bool {
64    match level.to_ascii_lowercase().as_str() {
65        "error" | "err" => {
66            set_level(Level::Error);
67            true
68        }
69        "warn" | "warning" => {
70            set_level(Level::Warn);
71            true
72        }
73        "info" => {
74            set_level(Level::Info);
75            true
76        }
77        "debug" => {
78            set_level(Level::Debug);
79            true
80        }
81        _ => false,
82    }
83}

127#[cfg(feature = "file-logging")]
128#[must_use]
129/// Initialize file logging to a specific path. Returns `true` on success.
130pub fn init_file_logging(path: &std::path::Path) -> bool {
131    OpenOptions::new()
132        .create(true)
133        .append(true)
134        .open(path)
135        .is_ok_and(|file| {
136            LOG_FILE.lock().is_ok_and(|mut log_file| {
137                *log_file = Some(file);
138                true
139            })
140        })
141}

144/// Initialize file logging (no-op when `file-logging` feature is disabled).
145pub fn init_file_logging(_path: &std::path::Path) -> bool { false }

147#[cfg(feature = "file-logging")]
148fn write_to_file(message: &str) {
149    if let Ok(mut log_file) = LOG_FILE.lock() {
150        if let Some(ref mut file) = *log_file {
151            let _ = writeln!(file, "{message}");
152            let _ = file.flush();
153        }
154    }
155}
156
157#[cfg(not(feature = "file-logging"))]
158fn write_to_file(_message: &str) {}
159
160#[cfg(feature = "file-logging")]
161fn is_file_logging_active() -> bool { LOG_FILE.lock().map(|lf| lf.is_some()).unwrap_or(false) }
162#[cfg(not(feature = "file-logging"))]
163fn is_file_logging_active() -> bool { false }

165fn emit(prefix: &str, msg: &str, to_stderr: bool) {
166    #[cfg(feature = "file-logging")]
167    {
168        if is_file_logging_active() && !prefix.is_empty() {
169            let file_message = format!("{prefix} {msg}");
170            write_to_file(&file_message);
171            return;
172        }
173    }
174    #[cfg(target_arch = "wasm32")]
175    {
176        let _ = to_stderr;
177        if prefix.is_empty() { console::log_1(&JsValue::from_str(msg)); }
178        else {
179            let formatted = format!("%c{} {}", prefix, msg);
180            fn style_for(prefix: &str) -> &'static str {
181                match prefix {
182                    "[ERROR]" => "color:#fff;background:#c0392b;font-weight:bold;padding:1px 4px;border-radius:3px",
183                    "[WARN]" => "color:#000;background:#ffeb3b;font-weight:bold;padding:1px 4px;border-radius:3px",
184                    "[INFO]" => "",
185                    "[DEBUG]" => "color:#000;background:#bdc3c7;padding:1px 4px;border-radius:3px",
186                    _ => "font-weight:bold",
187                }
188            }
189            let style = style_for(prefix);
190            let formatted_js = JsValue::from_str(&formatted);
191            let style_js = JsValue::from_str(style);
192            match prefix {
193                "[ERROR]" => console::error_2(&formatted_js, &style_js),
194                "[WARN]" => console::warn_2(&formatted_js, &style_js),
195                _ => console::log_2(&formatted_js, &style_js),
196            }
197        }
198    }
199    #[cfg(not(target_arch = "wasm32"))]
200    {
201        if to_stderr {
202            if prefix.is_empty() { eprintln!("{msg}"); }
203            else { eprintln!("{prefix} {msg}"); }
204        } else if prefix.is_empty() { println!("{msg}"); }
205        else { println!("{prefix} {msg}"); }
206    }
207}

209fn should_log(level: Level) -> bool {
210    match level {
211        Level::Info => { if !cfg!(feature = "log-info") { return false; } }
212        Level::Debug => { if !cfg!(feature = "log-debug") { return false; } }
213        _ => {}
214    }
215    let current = LOG_LEVEL.load(Ordering::SeqCst);
216    (level as u8) <= current && (level != Level::Debug || is_debug_enabled())
217}

219/// Internal logging dispatcher used by public macros.
220pub fn log_impl(level: Level, args: Arguments) {
221    if !should_log(level) { return; }
222    let msg = args.to_string();
223    match level {
224        Level::Error => emit("[ERROR]", &msg, true),
225        Level::Warn => emit("[WARN]", &msg, true),
226        Level::Info => emit("[INFO]", &msg, false),
227        Level::Debug => emit("[DEBUG]", &msg, false),
228    }
229}

231#[macro_export]
232/// Logs an error-level message (always enabled).
233macro_rules! error { ($($arg:tt)*) => { $crate::logger::log_impl($crate::logger::Level::Error, format_args!($($arg)*)) }; }
234#[macro_export]
235/// Logs a warning-level message (always enabled).
236macro_rules! warn  { ($($arg:tt)*) => { $crate::logger::log_impl($crate::logger::Level::Warn,  format_args!($($arg)*)) }; }
237#[macro_export]
238/// Logs an info-level message (requires `log-info` feature).
239macro_rules! info  { ($($arg:tt)*) => { $crate::logger::log_impl($crate::logger::Level::Info,  format_args!($($arg)*)) }; }
240#[macro_export]
241/// Logs a debug-level message (requires `log-debug` feature and runtime enablement).
242macro_rules! debug { ($($arg:tt)*) => { $crate::logger::log_impl($crate::logger::Level::Debug, format_args!($($arg)*)) }; }
243#[macro_export]
244/// Prints a verbose message (requires `verbose` feature and runtime enablement). This does not write to log files.
245macro_rules! verbose {
246    ($($arg:tt)*) => {
247        #[cfg(feature = "verbose")]
248        {
249            if $crate::logger::is_verbose_enabled() { println!($($arg)*); }
250        }
251    }
